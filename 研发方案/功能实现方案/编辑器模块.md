## 1. 模块概述

### 1.1 模块定位

编辑器模块是MiniAI记事本的核心功能模块，负责笔记的创建和编辑操作。该模块采用Neo-Brutalist设计风格，提供沉浸式的编辑体验。

### 1.2 模块职责

- 笔记内容的输入与编辑
- 标签关联管理
- 卡片外观定制
- 编辑历史控制（撤销/重做）
- 笔记保存与持久化

### 1.3 设计规范遵循

| 规范项 | 值 | 说明 |
|--------|-----|------|
| 主背景色 | #131313 | 深黑色 |

#### 3.1.4 状态管理

```typescript
// composables/useEditorHistory.ts
import { ref, computed } from 'vue'
import type { Note } from '@/types/entities'

interface HistoryState {
  past: Note[]
  present: Note
  future: Note[]
}

export function useEditorHistory(initialState: Note) {
  const history = ref<HistoryState>({
    past: [],
    present: initialState,
    future: []
  })

  const canUndo = computed(() => history.value.past.length > 0)
  const canRedo = computed(() => history.value.future.length > 0)

  function undo() {
    if (!canUndo.value) return
    
    const previous = history.value.past.pop()!
    history.value.future.unshift(history.value.present)
    history.value.present = previous
  }

  function redo() {
    if (!canRedo.value) return
    
    const next = history.value.future.shift()!
    history.value.past.push(history.value.present)
    history.value.present = next
  }

  function pushState(state: Note) {
    history.value.past.push(history.value.present)
    history.value.present = state
    history.value.future = []
  }

  function reset(state: Note) {
    history.value = {
      past: [],
      present: state,
      future: []
    }
  }

  return {
    history,
    canUndo,
    canRedo,
    undo,
    redo,
    pushState,
    reset
  }
}
```

#### 3.2.4 Props与事件定义

```typescript
interface Props {
  modelValue: string
  maxLength?: number
}

interface Emits {
  (e: 'update:modelValue', value: string): void
  (e: 'change', value: string): void
}

const props = withDefaults(defineProps<Props>(), {
  maxLength: 50
})

const emit = defineEmits<Emits>()

const title = computed({
  get: () => props.modelValue,
  set: (value) => emit('update:modelValue', value)
})

function handleTitleChange() {
  emit('change', title.value)
}
```

```typescript
// 标题验证逻辑

    return 'Title is required'
  }

    return `Title must be less than ${props.maxLength} characters`
  }
  return null
})
```

```typescript
// composables/useTextFormat.ts
export function useTextFormat(content: Ref<string>) {
  function insertBold() {
    insertFormatting('**', '**')
  }

  function insertItalic() {
    insertFormatting('*', '*')
  }

  function insertList() {
    const listItem = '\n- '
    content.value += listItem
  }

  function insertImage() {
    const imageTemplate = '\n![image description](url)'
    content.value += imageTemplate
  }

  function insertLink() {
    const linkTemplate = '[link text](url)'
    content.value += linkTemplate
  }

  function insertFormatting(before: string, after: string) {
    content.value += `${before}text${after}`
  }

  return {
    insertBold,
    insertItalic,
    insertList,
    insertImage,
    insertLink
  }
}
```

```typescript
// pages/notes/EditNotePage.vue
<script setup lang="ts">
import { ref, computed, onMounted, watch } from 'vue'
import { useRouter, useRoute } from 'vue-router'
import { useNoteStore } from '@/stores/modules/noteStore'
import { useEditorHistory } from '@/composables/useEditorHistory'
import { useResponsive } from '@/composables/useResponsive'
import { useTextFormat } from '@/composables/useTextFormat'
import { useToast } from '@/composables/useToast'
import type { Note, NoteDraft } from '@/types/entities'

const router = useRouter()
const route = useRoute()
const noteStore = useNoteStore()
const { isMobile } = useResponsive()
const { showToast } = useToast()

// 编辑状态
const isSaving = ref(false)
const isNewNote = computed(() => !route.params.id)

// 笔记草稿数据
const noteDraft = ref<NoteDraft>({
  title: '',
  content: '',
  category: 'Tasks',
  tags: [],
  cardColor: 'yellow',
  fontWeight: 700,
  reminder: null
})

// 历史管理
const { 
  history, 
  canUndo, 
  canRedo, 
  undo, 
  redo, 
  pushState, 
  reset 
} = useEditorHistory(noteDraft.value)

// 可用标签
const availableTags = computed(() => noteStore.availableTags)

// 最后编辑时间
const lastEditedTime = ref('2 mins ago')

// 验证状态
const isValid = computed(() => 
  noteDraft.value.title.trim().length > 0 && 
  noteDraft.value.content.trim().length > 0
)

// 加载现有笔记
onMounted(async () => {
  if (route.params.id) {
    const note = await noteStore.fetchNoteById(route.params.id as string)
    if (note) {
      noteDraft.value = { ...note }
      reset(noteDraft.value)
    }
  }
})

// 自动保存历史
watch(noteDraft, (newDraft) => {
  pushState(newDraft)
}, { deep: true })

// 处理函数
function handleClose() {
  if (!isValid.value) {
    router.push('/notes')
    return
  }
  // 提示保存
  showConfirmDialog('Save changes before closing?', {
    onConfirm: handleSave,
    onCancel: () => router.push('/notes')
  })
}

async function handleSave() {
  if (!isValid.value || isSaving.value) return
  
  isSaving.value = true
  try {
    if (isNewNote.value) {
      await noteStore.createNote(noteDraft.value)
      showToast('Note created successfully', 'success')
    } else {
      await noteStore.updateNote(route.params.id as string, noteDraft.value)
      showToast('Note updated successfully', 'success')
    }
    router.push('/notes')
  } catch (error) {
    showToast('Failed to save note', 'error')
  } finally {
    isSaving.value = false
  }
}

function handleToolbarSave() {
  handleSave()
}

function handleUndo() {
  undo()
  noteDraft.value = history.value.present
}

function handleRedo() {
  redo()
  noteDraft.value = history.value.present
}

function handleAddTag(tagName: string) {
  if (!noteDraft.value.tags.includes(tagName)) {
    noteDraft.value.tags.push(tagName)
  }
}

function handleRemoveTag(tagId: string) {
  noteDraft.value.tags = noteDraft.value.tags.filter(id => id !== tagId)
}

function handleColorChange(color: CardColor) {
  noteDraft.value.cardColor = color
}

function handleFontWeightChange(weight: number) {
  noteDraft.value.fontWeight = weight
}

function handleSetReminder(datetime: Date) {
  noteDraft.value.reminder = datetime.toISOString()
}

// 文本格式化
const { insertBold, insertItalic, insertList, insertImage, insertLink } = 
  useTextFormat(computed(() => noteDraft.value.content))
</script>
```

## 10. 技术要点总结

### 10.1 Neo-Brutalist设计实现要点

1. **无圆角设计**: 所有组件 border-radius: 0
2. **粗边框**: 4px solid white 统一边框
3. **偏移阴影**: 8px 8px 0 阴影效果
4. **Hover上移**: translate(-4px, -4px) + 阴影增强
5. **黄色强调**: Focus状态、FAB、保存按钮使用 #FFD700
6. **青色阴影**: 部分组件使用 #007F7F 作为阴影色

### 10.2 响应式适配要点

1. **Tailwind断点**: md (768px), lg (1024px)
2. **Grid布局**: grid-cols-1 lg:grid-cols-12
3. **条件渲染**: md:hidden 控制移动端组件
4. **字体大小**: text-3xl md:text-5xl

### 10.3 状态管理要点

1. **Pinia Store**: noteStore 管理笔记数据
2. **Composables**: useEditorHistory 管理编辑历史
3. **双向绑定**: v-model 实现标题和内容实时更新
4. **自动保存**: watch 监听变化自动push历史
