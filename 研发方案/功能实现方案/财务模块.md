# MiniAI记事本 - 财务模块功能实现方案

## 文档信息
- **项目名称：** KINETIC_NOTES（MiniAI记事本）
- **设计风格：** Neo-Brutalist
- **创建日期：** 2026-04-20

---

## 1. 模块概述

### 1.1 功能定位
财务模块提供个人收支记录与统计功能：
- **收入记录**：记录收入来源、金额、日期
- **支出记录**：记录支出分类、金额、日期
- **月度统计**：收支对比、分类占比可视化
- **趋势分析**：近6个月收支趋势图表

### 1.2 设计系统应用

| Neo-Brutalist元素 | 本模块应用 |
|-------------------|-----------|
| **零圆角** | 所有卡片、按钮 `rounded-none` |
| **粗边框** | 统计卡片 `border-4 border-white` |
| **偏移阴影** | 数字卡片 `shadow-[8px_8px_0px_0px_rgba(0,0,0,1)]` |
| **高对比配色** | 收入青色(#007F7F)，支出金色(#FFD700) |
| **数值强调** | 大号字体 `text-5xl font-black` |

### 3.4 CategoryPieChart 分类饼图

#### Neo-Brutalist饼图实现
```vue
<!-- components/business/CategoryPieChart.vue -->
<script setup lang="ts">
interface CategoryData {
  category: string
  amount: number
  percentage: number
  color: string
}

interface Props {
  data: CategoryData[]
  size?: number
}

const props = withDefaults(defineProps<Props>(), {
  size: 200,
})

// 计算饼图扇形路径
const calculateArcPath = (startAngle: number, endAngle: number, radius: number) => {
  const start = polarToCartesian(radius, radius, radius, startAngle)
  const end = polarToCartesian(radius, radius, radius, endAngle)
  const largeArc = endAngle - startAngle > 180 ? 1 : 0

  return [
    `M ${radius} ${radius}`,
    `L ${start.x} ${start.y}`,
    `A ${radius} ${radius} 0 ${largeArc} 1 ${end.x} ${end.y}`,
    'Z',
  ].join(' ')
}

const polarToCartesian = (cx: number, cy: number, r: number, angle: number) => {
  const radians = (angle - 90) * Math.PI / 180
  return {
    x: cx + r * Math.cos(radians),
    y: cy + r * Math.sin(radians),
  }
}

// 计算每个扇形的起始和结束角度
const arcs = computed(() => {
  let currentAngle = 0
  return props.data.map(item => {
    const startAngle = currentAngle
    const endAngle = currentAngle + (item.percentage * 360 / 100)
    currentAngle = endAngle
    return {
      path: calculateArcPath(startAngle, endAngle, props.size / 2),
      ...item,
    }
  })
})
</script>

<template>
  <div class="p-8 bg-surfaceLowest border-4 border-white">
    <!-- 标题 -->
    <h3 class="
      font-headline font-bold text-xs
      uppercase tracking-widest text-white/60 mb-8
    ">
      EXPENSE BREAKDOWN
    </h3>

    <!-- 饼图 -->
    <div class="flex items-center gap-8">
      <svg
        :width="size"
        :height="size"
        class="border-4 border-white rounded-none"
      >
        <g>
          <path
            v-for="arc in arcs"
            :key="arc.category"
            :d="arc.path"
            :fill="arc.color"
            class="stroke-white stroke-2 hover:opacity-80 transition-opacity"
          />
        </g>
      </svg>

      <!-- 分类列表 -->
      <div class="flex flex-col gap-3">
        <div
          v-for="item in data"
          :key="item.category"
          class="flex items-center gap-4"
        >
          <div
            class="w-4 h-4 border-2 border-white rounded-none"
            :style="{ backgroundColor: item.color }"
          />
          <span class="font-headline text-sm text-white uppercase">
            {{ item.category }}
          </span>
          <span class="font-headline font-bold text-sm text-primary">
            {{ item.percentage }}%
          </span>
        </div>
      </div>
    </div>
  </div>
</template>
```

## 4. 状态管理设计

### 4.1 Pinia Store

```typescript
// stores/finance.store.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { incomeRepo, expenseRepo } from '@/repositories'

export const useFinanceStore = defineStore('finance', () => {
  // ========== State ==========
  const incomes = ref<Income[]>([])
  const expenses = ref<Expense[]>([])
  const selectedMonth = ref<string>(getCurrentMonth())
  const loading = ref(false)

  // ========== Computed ==========
  // 当月总收入
  const monthlyIncome = computed(() => {
    return incomes.value
      .filter(i => i.date.startsWith(selectedMonth.value))
      .reduce((sum, i) => sum + i.amount, 0)
  })

  // 当月总支出
  const monthlyExpense = computed(() => {
    return expenses.value
      .filter(e => e.date.startsWith(selectedMonth.value))
      .reduce((sum, e) => sum + e.amount, 0)
  })

  // 当月结余
  const monthlyBalance = computed(() => {
    return monthlyIncome.value - monthlyExpense.value
  })

  // 支出分类统计
  const expenseByCategory = computed(() => {
    const monthlyExpenses = expenses.value.filter(
      e => e.date.startsWith(selectedMonth.value)
    )
    const grouped = new Map<string, number>()
    for (const exp of monthlyExpenses) {
      const current = grouped.get(exp.category) || 0
      grouped.set(exp.category, current + exp.amount)
    }
    return grouped
  })

  // 6个月趋势数据
  const trendData = computed(() => {
    const months = getLast6Months()
    return months.map(month => ({
      month,
      income: incomes.value
        .filter(i => i.date.startsWith(month))
        .reduce((sum, i) => sum + i.amount, 0),
      expense: expenses.value
        .filter(e => e.date.startsWith(month))
        .reduce((sum, e) => sum + e.amount, 0),
    }))
  })

  // ========== Actions ==========
  async function loadData() {
    loading.value = true
    try {
      incomes.value = await incomeRepo.findAll()
      expenses.value = await expenseRepo.findAll()
    } finally {
      loading.value = false
    }
  }

  async function addIncome(income: Omit<Income, 'id' | 'created_at' | 'updated_at' | 'sync_status' | 'is_deleted' | 'local_version'>) {
    await incomeRepo.create(income)
    await loadData()
  }

  async function addExpense(expense: Omit<Expense, 'id' | 'created_at' | 'updated_at' | 'sync_status' | 'is_deleted' | 'local_version'>) {
    await expenseRepo.create(expense)
    await loadData()
  }

  async function updateIncome(id: string, changes: Partial<Income>) {
    await incomeRepo.update(id, changes)
    await loadData()
  }

  async function updateExpense(id: string, changes: Partial<Expense>) {
    await expenseRepo.update(id, changes)
    await loadData()
  }

  async function deleteIncome(id: string) {
    await incomeRepo.delete(id)
    await loadData()
  }

  async function deleteExpense(id: string) {
    await expenseRepo.delete(id)
    await loadData()
  }

  function setMonth(month: string) {
    selectedMonth.value = month
  }

  async function initialize() {
    await loadData()
  }

  return {
    incomes,
    expenses,
    selectedMonth,
    loading,
    monthlyIncome,
    monthlyExpense,
    monthlyBalance,
    expenseByCategory,
    trendData,
    loadData,
    addIncome,
    addExpense,
    updateIncome,
    updateExpense,
    deleteIncome,
    deleteExpense,
    setMonth,
    initialize,
  }
})

// 辅助函数
function getCurrentMonth(): string {
  const now = new Date()
  return `${now.getFullYear()}-${String(now.getMonth() + 1).padStart(2, '0')}`
}

function getLast6Months(): string[] {
  const months: string[] = []
  const now = new Date()
  for (let i = 5; i >= 0; i--) {
    const d = new Date(now.getFullYear(), now.getMonth() - i, 1)
    months.push(`${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`)
  }
  return months
}
```

## 6. Repository扩展

### 6.1 FinanceRepository
```typescript
// repositories/finance.repository.ts
class FinanceRepository {
  // ========== 收入操作 ==========
  async getIncomeByMonth(month: string): Promise<Income[]> {
    return db.incomes
      .where('date')
      .startsWith(month)
      .and(income => !income.is_deleted)
      .toArray()
  }

  async getIncomeBySource(source: string): Promise<Income[]> {
    return db.incomes
      .where('source')
      .equals(source)
      .and(income => !income.is_deleted)
      .toArray()
  }

  async getTotalIncome(month: string): Promise<number> {
    const incomes = await this.getIncomeByMonth(month)
    return incomes.reduce((sum, i) => sum + i.amount, 0)
  }

  // ========== 支出操作 ==========
  async getExpenseByMonth(month: string): Promise<Expense[]> {
    return db.expenses
      .where('date')
      .startsWith(month)
      .and(expense => !expense.is_deleted)
      .toArray()
  }

  async getExpenseByCategory(category: string): Promise<Expense[]> {
    return db.expenses
      .where('category')
      .equals(category)
      .and(expense => !expense.is_deleted)
      .toArray()
  }

  async getTotalExpense(month: string): Promise<number> {
    const expenses = await this.getExpenseByMonth(month)
    return expenses.reduce((sum, e) => sum + e.amount, 0)
  }

  // ========== 统计分析 ==========
  async getCategorySummary(month: string): Promise<Map<string, number>> {
    const expenses = await this.getExpenseByMonth(month)
    const summary = new Map<string, number>()
    for (const exp of expenses) {
      const current = summary.get(exp.category) || 0
      summary.set(exp.category, current + exp.amount)
    }
    return summary
  }

  async getSourceSummary(month: string): Promise<Map<string, number>> {
    const incomes = await this.getIncomeByMonth(month)
    const summary = new Map<string, number>()
    for (const inc of incomes) {
      const current = summary.get(inc.source) || 0
      summary.set(inc.source, current + inc.amount)
    }
    return summary
  }

  async getTrendData(months: string[]): Promise<MonthTrend[]> {
    const trends: MonthTrend[] = []
    for (const month of months) {
      trends.push({
        month,
        income: await this.getTotalIncome(month),
        expense: await this.getTotalExpense(month),
      })
    }
    return trends
  }
}

export const financeRepo = new FinanceRepository()
```

```vue
  if (confirm('Delete this record?')) {

    }
  }

  const data: CategoryData[] = []
  for (const [category, amount] of financeStore.expenseByCategory) {
    data.push({
      category,
      amount,
      percentage: (amount / total) * 100,
      color: CATEGORY_COLORS[category] || '#353535',
    })
  }
  return data.sort((a, b) => b.amount - a.amount)
})
```
