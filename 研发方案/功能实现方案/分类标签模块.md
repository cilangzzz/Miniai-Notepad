# MiniAI记事本 - 分类标签模块功能实现方案

## 文档信息
- **项目名称：** KINETIC_NOTES（MiniAI记事本）
- **设计风格：** Neo-Brutalist
- **创建日期：** 2026-04-20

---

## 1. 模块概述

### 1.1 功能定位
分类标签模块是记事本的核心组织系统，提供：
- **分类管理**：Work/Personal/Ideas/Tasks四大预设分类
- **标签管理**：用户自定义标签，支持旋转视觉效果
- **筛选导航**：通过分类/标签快速定位笔记
- **统计展示**：各分类笔记数量可视化

### 1.2 设计系统应用

| Neo-Brutalist元素 | 本模块应用 |
|-------------------|-----------|
| **零圆角** | 所有卡片、标签使用 `rounded-none` |
| **粗边框** | 分类卡片 `border-4 border-white`，标签 `border-2` |
| **偏移阴影** | 卡片阴影 `8px 8px 0px 0px rgba(0,0,0,1)` |
| **倾斜变换** | 标签随机旋转 `-rotate-3` 到 `rotate-3` |
| **高对比配色** | 激活分类使用 `bg-secondary` (#007F7F) |

### 3.2 TagCloud 标签云组件

#### 组件规格
```vue
<!-- components/business/TagCloud.vue -->
<script setup lang="ts">
interface Tag {
  id: string
  name: string
  color: 'gold' | 'teal' | 'white' | 'gray'
  rotation: number  // -3 到 3
  usageCount: number
  selected?: boolean
}

interface Props {
  tags: Tag[]
  selectedIds?: string[]
  editable?: boolean
}

const props = withDefaults(defineProps<Props>(), {
  selectedIds: [],
  editable: true,
})

const emit = defineEmits<{
  select: [tagId: string]
  create: []
  remove: [tagId: string]
}>()

// 计算每个标签的旋转角度（随机分配）
const getRotation = (index: number) => {
  const rotations = [-3, -2, -1, 0, 1, 2, 3]
  return rotations[index % rotations.length]
}
</script>

<template>
  <div class="relative p-8 bg-surfaceLowest border-4 border-white">
    <!-- 标题栏 -->
    <div class="flex justify-between items-center mb-8">
      <h2 class="
        font-headline font-black text-xl
        uppercase tracking-tighter text-primary
        -skew-x-2
      ">
        TAGS
      </h2>

      <button
        v-if="editable"
        @click="emit('create')"
        class="
          px-4 py-2
          bg-secondary border-2 border-white
          text-white font-headline font-bold uppercase
          hover:bg-primary hover:text-black
          transition-all duration-150
        "
      >
        <span class="material-symbols-outlined text-sm">add</span>
        NEW TAG
      </button>
    </div>

    <!-- 标签网格 -->
    <div class="flex flex-wrap gap-4 items-center">
      <NeoTag
        v-for="(tag, index) in tags"
        :key="tag.id"
        :label="tag.name"
        :color="tag.color"
        :rotation="getRotation(index)"
        :selected="selectedIds.includes(tag.id)"
        :removable="editable"
        @click="emit('select', tag.id)"
        @remove="emit('remove', tag.id)"
      />
    </div>

    <!-- 空状态 -->
    <div
      v-if="tags.length === 0"
      class="
        py-12 text-center
        text-white/50 font-headline
      "
    >
      <span class="material-symbols-outlined text-4xl mb-4">label_off</span>
      <p class="text-sm uppercase">No tags yet</p>
    </div>
  </div>
</template>
```

## 4. 状态管理设计

### 4.1 Pinia Store

```typescript
// stores/category.store.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { categoryRepo, tagRepo, noteRepo } from '@/repositories'

export const useCategoryStore = defineStore('category', () => {
  // ========== State ==========
  const categories = ref<Category[]>([])
  const tags = ref<Tag[]>([])
  const activeCategory = ref<string | null>(null)
  const selectedTags = ref<string[]>([])
  const loading = ref(false)

  // ========== Computed ==========
  const categoryStats = computed(() => {
    const stats = new Map<string, number>()
    for (const cat of categories.value) {
      stats.set(cat.name, cat.note_count)
    }
    return stats
  })

  const totalNotes = computed(() =>
    categories.value.reduce((sum, cat) => sum + cat.note_count, 0)
  )

  const filteredTags = computed(() => {
    if (selectedTags.value.length === 0) return tags.value
    return tags.value.filter(tag => selectedTags.value.includes(tag.id))
  })

  // ========== Actions ==========
  async function loadCategories() {
    loading.value = true
    try {
      categories.value = await categoryRepo.findAll()
    } finally {
      loading.value = false
    }
  }

  async function loadTags() {
    loading.value = true
    try {
      tags.value = await tagRepo.findAll()
    } finally {
      loading.value = false
    }
  }

  async function selectCategory(categoryId: string) {
    activeCategory.value = categoryId
    // 触发笔记筛选
    await noteRepo.findByCategory(categoryId)
  }

  async function toggleTag(tagId: string) {
    const index = selectedTags.value.indexOf(tagId)
    if (index === -1) {
      selectedTags.value.push(tagId)
    } else {
      selectedTags.value.splice(index, 1)
    }
  }

  async function createTag(name: string, color: CardColor) {
    const randomRotation = Math.floor(Math.random() * 7) - 3
    await tagRepo.create({
      name,
      color,
      rotation: randomRotation,
      usage_count: 0,
      sync_status: 'local',
      is_deleted: false,
    })
    await loadTags()
  }

  async function deleteTag(tagId: string) {
    await tagRepo.delete(tagId)
    selectedTags.value = selectedTags.value.filter(id => id !== tagId)
    await loadTags()
  }

  async function refreshStats() {
    for (const cat of categories.value) {
      const notes = await noteRepo.findByCategory(cat.id)
      await categoryRepo.update(cat.id, { note_count: notes.length })
    }
    await loadCategories()
  }

  // ========== 初始化 ==========
  async function initialize() {
    await loadCategories()
    await loadTags()
  }

  return {
    categories,
    tags,
    activeCategory,
    selectedTags,
    loading,
    categoryStats,
    totalNotes,
    filteredTags,
    loadCategories,
    loadTags,
    selectCategory,
    toggleTag,
    createTag,
    deleteTag,
    refreshStats,
    initialize,
  }
})
```

## 6. Repository扩展

### 6.1 CategoryRepository
```typescript
// repositories/category.repository.ts
class CategoryRepository extends BaseRepository<Category> {
  constructor() {
    super(db.categories)
  }

  async findBySlug(slug: string): Promise<Category | null> {
    return this.table.where('slug').equals(slug).first()
  }

  async updateNoteCount(categoryId: string): Promise<void> {
    const count = await db.notes
      .where('category_id')
      .equals(categoryId)
      .and(note => !note.is_archived && !note.is_deleted)
      .count()

    await this.update(categoryId, { note_count: count })
  }

  async getStats(): Promise<CategoryStats[]> {
    const categories = await this.findAll()
    return categories.map(cat => ({
      id: cat.id,
      name: cat.name,
      color: cat.color,
      noteCount: cat.note_count,
    }))
  }
}

export const categoryRepo = new CategoryRepository()
```

### 6.2 TagRepository
```typescript
// repositories/tag.repository.ts
class TagRepository extends BaseRepository<Tag> {
  constructor() {
    super(db.tags)
  }

  async findByName(name: string): Promise<Tag | null> {
    return this.table.where('name').equals(name).first()
  }

  async incrementUsage(tagId: string): Promise<void> {
    const tag = await this.findById(tagId)
    if (tag) {
      await this.update(tagId, { usage_count: tag.usage_count + 1 })
    }
  }

  async decrementUsage(tagId: string): Promise<void> {
    const tag = await this.findById(tagId)
    if (tag && tag.usage_count > 0) {
      await this.update(tagId, { usage_count: tag.usage_count - 1 })
    }
  }

  async getPopularTags(limit: number = 10): Promise<Tag[]> {
    return this.table
      .orderBy('usage_count')
      .reverse()
      .limit(limit)
      .toArray()
  }
}

export const tagRepo = new TagRepository()
```
