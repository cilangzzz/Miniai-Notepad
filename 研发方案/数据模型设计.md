## 1. 设计原则

### 1.1 云迁移友好设计
- **存储无关性：** Repository抽象层屏蔽底层存储实现
- **同步预留字段：** 所有实体包含云同步必需字段
- **版本控制：** 支持乐观锁，冲突检测
- **增量同步：** 通过时间戳追踪变更

### 1.2 IndexedDB适配
- **主键设计：** 使用UUID作为主键，便于云迁移
- **索引设计：** 常用查询字段建立索引
- **事务支持：** 复合操作使用事务保证一致性

## 2. 基础实体接口

### 2.1 SyncableEntity（同步实体基类）

```typescript
/**
 * 同步实体基类 - 所有需要云同步的实体都继承此接口
 */
interface SyncableEntity {
  // ========== 本地标识 ==========
  id: string              // UUID，本地生成的主键

  // ========== 云同步预留字段 ==========
  cloud_id?: string       // 云端唯一标识（同步后填充）
  sync_status: SyncStatus // 同步状态
  local_version: number   // 本地数据版本（乐观锁）
  cloud_version?: number  // 云端数据版本（冲突检测）

  // ========== 时间戳 ==========
  created_at: number      // 创建时间戳 (milliseconds)
  updated_at: number      // 最后更新时间戳
  last_sync_at?: number   // 最后同步时间戳

  // ========== 删除标记 ==========
  is_deleted: boolean     // 软删除标记
  deleted_at?: number     // 删除时间戳
}

/**
 * 同步状态枚举
 */
type SyncStatus = 
  | 'local'     // 仅本地存在，未同步
  | 'pending'   // 待同步（本地变更未推送）
  | 'synced'    // 已同步（本地与云端一致）
  | 'conflict'  // 冲突（需要用户决定）
```

### 2.2 数据版本控制策略

```typescript
/**
 * 版本控制策略
 * - 乐观锁：通过version字段检测并发修改
 * - 冲突解决：Last-Write-Wins（最后写入优先）
 */
interface VersionControl {
  local_version: number   // 每次本地修改自增
  cloud_version?: number  // 云端版本号
  
  // 冲突检测逻辑
  hasConflict(): boolean {
    return this.cloud_version !== undefined 
      && this.local_version > this.cloud_version
  }
}
```

## 4. IndexedDB Schema设计

### 4.1 数据库定义

```typescript
import Dexie, { Table } from 'dexie'

/**
 * MiniAI记事本数据库
 */
class MiniAIDB extends Dexie {
  notes!: Table<Note, string>
  categories!: Table<Category, string>
  tags!: Table<Tag, string>
  incomes!: Table<Income, string>
  expenses!: Table<Expense, string>
  syncLog!: Table<SyncLogEntry, string>  // 同步日志（预留）

  constructor() {
    super('MiniAIDB')
    
    // 版本1：初始Schema
    this.version(1).stores({
      // 笔记表：主键id + 索引
      notes: 'id, cloud_id, category_id, is_archived, is_pinned, created_at, updated_at, [category_id+is_archived]',
      
      // 分类表：主键id + 索引
      categories: 'id, cloud_id, slug, sort_order',
      
      // 标签表：主键id + 索引
      tags: 'id, cloud_id, name, usage_count',
      
      // 收入表：主键id + 索引
      incomes: 'id, cloud_id, date, source, [date+source]',
      
      // 支出表：主键id + 索引
      expenses: 'id, cloud_id, date, category, [date+category]',
      
      // 同步日志表（预留）
      syncLog: 'id, entity_type, entity_id, operation, timestamp'
    })
  }
}

// 数据库实例
export const db = new MiniAIDB()
```

## 5. Repository抽象层设计

### 5.1 Repository接口定义

```typescript
/**
 * Repository通用接口
 */
interface IRepository<T extends SyncableEntity> {
  // ========== 基础CRUD ==========
  findById(id: string): Promise<T | null>
  findAll(): Promise<T[]>
  create(entity: Omit<T, 'id' | 'created_at' | 'updated_at' | 'local_version'>): Promise<T>
  update(id: string, changes: Partial<T>): Promise<T>
  delete(id: string): Promise<void>  // 软删除
  
  // ========== 批量操作 ==========
  bulkCreate(entities: T[]): Promise<T[]>
  bulkUpdate(ids: string[], changes: Partial<T>): Promise<T[]>
  bulkDelete(ids: string[]): Promise<void>

  // ========== 查询接口 ==========
  query(spec: QuerySpec): Promise<T[]>
  
  // ========== 同步预留接口 ==========
  getPendingSync(): Promise<T[]>           // 获取待同步实体
  markSynced(id: string, cloudId: string): Promise<void>
  resolveConflict(id: string, resolution: ConflictResolution): Promise<T>
}

/**
 * 查询规格
 */
interface QuerySpec {
  filters?: FilterCondition[]
  sort?: SortSpec
  limit?: number
  offset?: number
}

interface FilterCondition {
  field: string
  operator: 'eq' | 'neq' | 'gt' | 'gte' | 'lt' | 'lte' | 'in' | 'contains'
  value: any
}

interface SortSpec {
  field: string
  direction: 'asc' | 'desc'
}

/**
 * 冲突解决策略
 */
type ConflictResolution = 'keep_local' | 'keep_cloud' | 'merge'
```

### 5.2 BaseRepository实现

```typescript
/**
 * Repository基类实现
 */
abstract class BaseRepository<T extends SyncableEntity> implements IRepository<T> {
  protected table: Table<T, string>
  protected generateId: () => string

  constructor(table: Table<T, string>) {
    this.table = table
    this.generateId = () => crypto.randomUUID()
  }

  async findById(id: string): Promise<T | null> {
    return this.table.get(id) ?? null
  }

  async findAll(): Promise<T[]> {
    return this.table.toArray()
  }

  async create(entity: Omit<T, 'id' | 'created_at' | 'updated_at' | 'local_version'>): Promise<T> {
    const now = Date.now()
    const newEntity = {
      ...entity,
      id: this.generateId(),
      created_at: now,
      updated_at: now,
      local_version: 1,
      sync_status: 'local',
      is_deleted: false,
    } as T
    
    await this.table.add(newEntity)
    return newEntity
  }

  async update(id: string, changes: Partial<T>): Promise<T> {
    const existing = await this.findById(id)
    if (!existing) throw new Error(`Entity ${id} not found`)
    
    const now = Date.now()
    const updated = {
      ...existing,
      ...changes,
      updated_at: now,
      local_version: existing.local_version + 1,
      sync_status: existing.sync_status === 'synced' ? 'pending' : existing.sync_status,
    }
    
    await this.table.put(updated)
    return updated
  }

  async delete(id: string): Promise<void> {
    // 软删除
    await this.update(id, { 
      is_deleted: true, 
      deleted_at: Date.now(),
      sync_status: 'pending'
    })
  }

  async query(spec: QuerySpec): Promise<T[]> {
    let collection = this.table.toCollection()
    
    // 应用筛选条件
    if (spec.filters) {
      for (const filter of spec.filters) {
        collection = this.applyFilter(collection, filter)
      }
    }
    
    // 应用排序
    if (spec.sort) {
      // Dexie排序需要特殊处理
    }
    
    // 应用分页
    if (spec.offset) {
      collection = collection.offset(spec.offset)
    }
    if (spec.limit) {
      collection = collection.limit(spec.limit)
    }
    
    return collection.toArray()
  }

  // ========== 同步预留方法 ==========
  async getPendingSync(): Promise<T[]> {
    return this.table
      .where('sync_status')
      .equals('pending')
      .toArray()
  }

  async markSynced(id: string, cloudId: string): Promise<void> {
    await this.update(id, {
      cloud_id: cloudId,
      sync_status: 'synced',
      last_sync_at: Date.now()
    })
  }

  // 子类实现的筛选器
  protected abstract applyFilter(collection: any, filter: FilterCondition): any
}
```

### 5.3 具体Repository实现

```typescript
/**
 * 笔记Repository
 */
class NoteRepository extends BaseRepository<Note> {
  constructor() {
    super(db.notes)
  }

  // ========== 笔记特有查询 ==========
  async findByCategory(categoryId: string): Promise<Note[]> {
    return this.table
      .where('category_id')
      .equals(categoryId)
      .and(note => !note.is_archived && !note.is_deleted)
      .toArray()
  }

  async findArchived(): Promise<Note[]> {
    return this.table
      .where('is_archived')
      .equals(true)
      .and(note => !note.is_deleted)
      .toArray()
  }

  async findPinned(): Promise<Note[]> {
    return this.table
      .where('is_pinned')
      .equals(true)
      .and(note => !note.is_archived && !note.is_deleted)
      .toArray()
  }

  async searchByKeyword(keyword: string): Promise<Note[]> {
    const lowerKeyword = keyword.toLowerCase()
    return this.table
      .filter(note => 
        !note.is_deleted && !note.is_archived &&
        (note.title.toLowerCase().includes(lowerKeyword) ||
         note.content.toLowerCase().includes(lowerKeyword))
      )
      .toArray()
  }

  async archive(id: string): Promise<void> {
    await this.update(id, {
      is_archived: true,
      archived_at: Date.now()
    })
  }

  async unarchive(id: string): Promise<void> {
    await this.update(id, {
      is_archived: false,
      archived_at: undefined
    })
  }

  async pin(id: string): Promise<void> {
    await this.update(id, {
      is_pinned: true,
      pinned_at: Date.now()
    })
  }

  async unpin(id: string): Promise<void> {
    await this.update(id, {
      is_pinned: false,
      pinned_at: undefined
    })
  }

  protected applyFilter(collection: any, filter: FilterCondition): any {
    // 实现笔记特定的筛选逻辑
    return collection
  }
}

/**
 * 财务Repository
 */
class FinanceRepository extends BaseRepository<Income | Expense> {
  // ========== 财务统计查询 ==========
  async getMonthlySummary(type: 'income' | 'expense', month: string): Promise<number> {
    const table = type === 'income' ? db.incomes : db.expenses
    const records = await table
      .where('date')
      .startsWith(month)  // YYYY-MM
      .toArray()
    
    return records.reduce((sum, r) => sum + r.amount, 0)
  }

  async getCategorySummary(type: 'expense', month: string): Promise<Map<string, number>> {
    const records = await db.expenses
      .where('date')
      .startsWith(month)
      .toArray()
    
    const summary = new Map<string, number>()
    for (const record of records) {
      const current = summary.get(record.category) || 0
      summary.set(record.category, current + record.amount)
    }
    return summary
  }
}
```

### 7.1 Schema版本迁移

```typescript
/**
 * Schema迁移策略
 * - IndexedDB版本升级时执行迁移
 */
class SchemaMigration {
  static async migrate(db: MiniAIDB, oldVersion: number, newVersion: number) {
    switch (oldVersion) {
      case 0:
        // 初始版本，无需迁移
        break
      
      case 1:
        // 未来版本迁移逻辑
        await this.migrateV1ToV2(db)
        break
      
      default:
        console.warn(`Unknown version migration: ${oldVersion} -> ${newVersion}`)
    }
  }

  static async migrateV1ToV2(db: MiniAIDB) {
    // 示例：添加新字段
    const notes = await db.notes.toArray()
    for (const note of notes) {
      if (!note.font_weight) {
        await db.notes.update(note.id, { font_weight: 'normal' })
      }
    }
  }
}
```
