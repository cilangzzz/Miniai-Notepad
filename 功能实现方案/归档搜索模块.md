# MiniAI记事本 - 归档搜索模块功能实现方案

## 文档信息

- **项目名称：** KINETIC_NOTES（MiniAI记事本）
- **模块名称：** 归档搜索模块（Archive Search）
- **文档版本：** v1.0
- **创建日期：** 2026-04-20
- **设计风格：** Neo-Brutalist
- **状态：** 设计完成

---

## 1. 模块概述

### 1.1 功能定位

归档搜索模块提供已归档笔记的搜索、筛选和恢复功能，是用户管理历史记录的核心入口。

### 1.2 核心功能点

| 序号 | 功能点 | 组件名称 | 优先级 |
|------|--------|----------|--------|
| 1 | 搜索框 | ArchiveSearchBar | P0 |
| 2 | 筛选芯片 | FilterChips | P0 |
| 3 | 归档结果列表 | ArchiveResults | P0 |

### 1.3 技术栈

- **框架：** Vue 3 + Composition API + `<script setup>`
- **状态管理：** Pinia
- **样式系统：** TailwindCSS + 自定义Neo-Brutalist主题
- **数据存储：** IndexedDB (Dexie.js)

#### 3.1.5 组件模板实现

```vue
<!-- ArchiveSearchBar.vue -->
<script setup lang="ts">
import { ref, computed } from 'vue';

// Props定义
const props = withDefaults(defineProps<{
  placeholder?: string;
  initialValue?: string;
  disabled?: boolean;
  loading?: boolean;
  executeLabel?: string;
}>(), {
  placeholder: 'QUERY THE ARCHIVE...',
  initialValue: '',
  disabled: false,
  loading: false,
  executeLabel: 'EXECUTE'
});

// Events定义
const emit = defineEmits<{
  execute: [keyword: string];
  input: [keyword: string];
  clear: [];
  focus: [];
  blur: [];
}>();

// 本地状态
const searchKeyword = ref(props.initialValue);
const isFocused = ref(false);

// 计算属性：是否显示清空按钮
const showClear = computed(() => searchKeyword.value.length > 0);

// 计算属性：Execute按钮是否可用
const canExecute = computed(() => 
  !props.disabled && 
  !props.loading && 
  searchKeyword.value.length > 0
);

// 处理方法
const handleInput = (event: Event) => {
  const target = event.target as HTMLInputElement;
  searchKeyword.value = target.value;
  emit('input', target.value);
};

const handleExecute = () => {
  if (canExecute.value) {
    emit('execute', searchKeyword.value);
  }
};

const handleClear = () => {
  searchKeyword.value = '';
  emit('clear');
};

const handleFocus = () => {
  isFocused.value = true;
  emit('focus');
};

const handleBlur = () => {
  isFocused.value = false;
  emit('blur');
};

// 键盘事件：Enter执行搜索
const handleKeydown = (event: KeyboardEvent) => {
  if (event.key === 'Enter') {
    handleExecute();
  }
};
</script>

<template>
  <!-- 双层结构容器 -->
  <div class="neo-search-wrapper relative">
    
    <!-- 背景偏移层 -->
    <div 
      class="neo-search-bg absolute inset-0 bg-[#333333] border-4 border-white"
      aria-hidden="true"
    ></div>
    
    <!-- 主输入层 -->

      class="neo-search-main relative bg-[#131313] border-4 border-white"

    >
      <div class="flex items-stretch">
        
        <!-- 大号输入框 -->

          type="text"
          v-model="searchKeyword"
          :placeholder="props.placeholder"
          :disabled="props.disabled"
          class="neo-search-input flex-1 bg-transparent text-white font-medium text-lg px-6 py-6 md:px-8 md:py-8 focus:outline-none"
          :class="{ 'opacity-50': props.disabled }"
          @input="handleInput"
          @focus="handleFocus"
          @blur="handleBlur"
          @keydown="handleKeydown"
        />
        
        <!-- 清空按钮（可选显示） -->
        <button
          v-if="showClear"
          type="button"
          class="neo-clear-btn px-4 text-white/60 hover:text-white transition-colors"
          @click="handleClear"
          aria-label="Clear search"
        >
          <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
            <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12" />
          </svg>
        </button>
        
        <!-- Execute按钮 -->
        <button
          type="button"
          class="neo-execute-btn bg-[#FFD700] border-4 border-white px-8 font-bold uppercase tracking-wider"
          :class="{
            'shadow-[8px_8px_0_#007F7F] hover:shadow-[12px_12px_0_#007F7F] hover:-translate-x-1 hover:-translate-y-1': canExecute,
            'opacity-50 cursor-not-allowed': !canExecute,
            'animate-pulse': props.loading

            </svg>
            SEARCHING...
          </span>

      <!-- Focus状态青色阴影 -->
```

### 3.2 FilterChips（筛选芯片组件）

#### 3.2.6 组件模板实现

```vue
<!-- FilterChips.vue -->
<script setup lang="ts">
import { ref, computed, watch } from 'vue';
import type { TimeFilterOption, CategoryFilterOption } from '@/types/components';

// Props定义
const props = withDefaults(defineProps<{
  timeOptions?: TimeFilterOption[];
  categoryOptions?: CategoryFilterOption[];
  activeTimeFilter?: string;
  activeCategoryFilters?: string[];
  multiSelect?: boolean;
  showLabels?: boolean;
}>(), {
  timeOptions: () => TIME_FILTER_OPTIONS,
  categoryOptions: () => CATEGORY_FILTER_OPTIONS,
  activeTimeFilter: 'all',
  activeCategoryFilters: () => [],
  multiSelect: true,
  showLabels: true
});

// Events定义
const emit = defineEmits<{
  timeChange: [filterId: string];
  categoryChange: [filterIds: string[]];
  reset: [];
  chipClick: [type: 'time' | 'category', filterId: string];
}>();

// 本地状态
const localTimeFilter = ref(props.activeTimeFilter);
const localCategoryFilters = ref<string[]>([...props.activeCategoryFilters]);

// 监听props变化同步本地状态
watch(() => props.activeTimeFilter, (val) => {
  localTimeFilter.value = val;
});

watch(() => props.activeCategoryFilters, (val) => {
  localCategoryFilters.value = [...val];
}, { deep: true });

// 处理时间筛选点击
const handleTimeClick = (filterId: string) => {
  localTimeFilter.value = filterId;
  emit('timeChange', filterId);
  emit('chipClick', 'time', filterId);
};

// 处理分类筛选点击
const handleCategoryClick = (filterId: string) => {
  if (props.multiSelect) {
    // 多选模式：切换选中状态
    const index = localCategoryFilters.value.indexOf(filterId);
    if (index > -1) {
      localCategoryFilters.value.splice(index, 1);
    } else {
      localCategoryFilters.value.push(filterId);
    }
  } else {
    // 单选模式：直接替换
    localCategoryFilters.value = [filterId];
  }
  emit('categoryChange', [...localCategoryFilters.value]);
  emit('chipClick', 'category', filterId);
};

// 处理重置
const handleReset = () => {
  localTimeFilter.value = 'all';
  localCategoryFilters.value = [];
  emit('reset');
};

// 判断时间筛选是否激活
const isTimeActive = (filterId: string) => {
  return localTimeFilter.value === filterId;
};

// 判断分类筛选是否激活
const isCategoryActive = (filterId: string) => {
  return localCategoryFilters.value.includes(filterId);
};

// 计算激活数量
const activeCount = computed(() => {
  return localCategoryFilters.value.length + 
    (localTimeFilter.value !== 'all' ? 1 : 0);
});

// 获取分类颜色样式
const getCategoryColorClass = (color: string, isActive: boolean) => {
  if (isActive) {
    return 'bg-[#007F7F] text-white';  // 激活状态统一青色
  }
  // 根据分类颜色设置背景
  const colorMap: Record<string, string> = {
    yellow: 'bg-[#333333]',
    cyan: 'bg-[#333333]',
    white: 'bg-[#333333]',
    gray: 'bg-[#333333]',
  };
  return colorMap[color] || 'bg-[#333333]';
};
</script>

<template>
  <div class="neo-filter-chips">
    
    <!-- 筛选标题 -->
    <div v-if="props.showLabels" class="flex items-center justify-between mb-4">
      <h3 class="text-white font-bold uppercase tracking-wider text-sm" style="font-family: 'Space Grotesk', sans-serif;">
        FILTER BY
      </h3>
      
      <!-- 重置按钮 -->
      <button
        v-if="activeCount > 0"
        type="button"
        class="neo-reset-btn text-white/60 hover:text-white text-xs uppercase tracking-wider transition-colors"
        @click="handleReset"
      >
        RESET
      </button>
    </div>
    
    <!-- 时间筛选组 -->
    <div class="neo-filter-group mb-4">
      <div class="flex flex-wrap gap-2">
        <button
          v-for="option in props.timeOptions"
          :key="option.id"
          type="button"
          class="neo-chip px-4 py-2 border-2 border-white font-bold uppercase text-xs tracking-wider transition-all duration-150"
          :class="{
            'bg-[#007F7F] text-white shadow-[4px_4px_0_rgba(255,255,255,0.1)]': isTimeActive(option.id),
            'bg-[#131313] text-white hover:bg-[#007F7F]': !isTimeActive(option.id)
          }"
          @click="handleTimeClick(option.id)"
        >
          {{ option.label }}
        </button>
      </div>
    </div>
    
    <!-- 分类筛选组 -->
    <div class="neo-filter-group">
      <div class="flex flex-wrap gap-2">
        <button
          v-for="option in props.categoryOptions"
          :key="option.id"
          type="button"
          class="neo-chip px-4 py-2 border-2 border-white font-bold uppercase text-xs tracking-wider transition-all duration-150"
          :class="{
            'bg-[#007F7F] text-white shadow-[4px_4px_0_rgba(255,255,255,0.1)]': isCategoryActive(option.id),
            'bg-[#333333] text-white hover:bg-[#007F7F]': !isCategoryActive(option.id)
          }"
          @click="handleCategoryClick(option.id)"
        >
          <!-- 分类图标（可选） -->
          <span v-if="option.icon" class="mr-1">
            <!-- Material Symbol图标占位 -->
          </span>
          {{ option.label }}
        </button>
      </div>
    </div>
    
    <!-- 激活筛选数量指示 -->
    <div v-if="activeCount > 0" class="neo-filter-count mt-4 text-white/60 text-xs">
      {{ activeCount }} filter(s) active
    </div>
  </div>
</template>

<style scoped>
```

```vue
<!-- pages/ArchiveSearch.vue -->
<script setup lang="ts">
import { ref, computed, watch, onMounted } from 'vue';
import ArchiveSearchBar from '@/components/business/ArchiveSearchBar.vue';
import FilterChips from '@/components/business/FilterChips.vue';
import ArchiveResults from '@/components/business/ArchiveResults.vue';
import { noteRepo } from '@/repositories/note.repository';
import type { ArchivedNoteCard } from '@/types/components';

// ========== 状态管理 ==========
const searchKeyword = ref('');
const activeTimeFilter = ref('all');
const activeCategoryFilters = ref<string[]>([]);
const archivedNotes = ref<ArchivedNoteCard[]>([]);
const isLoading = ref(false);
const hasMore = ref(false);
const total = ref(0);

// ========== 搜索逻辑 ==========
const performSearch = async () => {
  isLoading.value = true;
  try {
    // 构建查询条件
    const filters: FilterCondition[] = [
      { field: 'is_archived', operator: 'eq', value: true },
      { field: 'is_deleted', operator: 'eq', value: false }
    ];
    
    // 添加关键词筛选
    if (searchKeyword.value) {
      filters.push({
        field: 'title',
        operator: 'contains',
        value: searchKeyword.value.toLowerCase()
      });
    }
    
    // 添加时间筛选
    if (activeTimeFilter.value !== 'all') {
      const daysMap: Record<string, number> = {
        '1d': 1,
        '7d': 7,
        '30d': 30
      };
      const days = daysMap[activeTimeFilter.value] || 0;
      if (days > 0) {
        const cutoffTime = Date.now() - days * 24 * 60 * 60 * 1000;
        filters.push({
          field: 'archived_at',
          operator: 'gte',
          value: cutoffTime
        });
      }
    }
    
    // 添加分类筛选
    if (activeCategoryFilters.value.length > 0) {
      filters.push({
        field: 'category_id',
        operator: 'in',
        value: activeCategoryFilters.value
      });
    }
    
    // 执行查询
    const result = await noteRepo.query({
      filters,
      sort: { field: 'archived_at', direction: 'desc' },
      limit: 20
    });
    
    // 转换结果格式
    archivedNotes.value = result.map(note => ({
      id: note.id,
      title: note.title,
      contentSummary: note.content,
      category: {
        id: note.category_id,
        name: note.category,
        color: note.card_color
      },
      tags: note.tags,
      cardType: note.card_type,
      archivedAt: note.archived_at || 0,
      createdAt: note.created_at
    }));
    
    total.value = result.length;
    hasMore.value = result.length === 20; // 假设有更多数据
  } finally {
    isLoading.value = false;
  }
};

// ========== 事件处理 ==========
const handleSearchExecute = (keyword: string) => {
  searchKeyword.value = keyword;
  performSearch();
};

const handleTimeChange = (filterId: string) => {
  activeTimeFilter.value = filterId;
  performSearch();
};

const handleCategoryChange = (filterIds: string[]) => {
  activeCategoryFilters.value = filterIds;
  performSearch();
};

const handleReset = () => {
  activeTimeFilter.value = 'all';
  activeCategoryFilters.value = [];
  performSearch();
};

const handleRestore = async (noteId: string) => {
  await noteRepo.unarchive(noteId);
  archivedNotes.value = archivedNotes.value.filter(n => n.id !== noteId);
  total.value--;
};

// ========== 初始化 ==========
onMounted(() => {
  performSearch();
});
```

```typescript
export const useArchiveStore = defineStore('archive', () => {

  const searchKeyword = ref('');

  const archivedNotes = ref<ArchivedNoteCard[]>([]);

  const hasFilters = computed(() =>
```

## 6. 数据查询方案

### 6.1 Repository查询方法扩展

```typescript
// repositories/note.repository.ts 扩展

/**
 * 归档查询方法
 */
async findArchived(options?: ArchiveQueryOptions): Promise<Note[]> {
  let collection = this.table
    .where('is_archived')
    .equals(true)
    .and(note => !note.is_deleted);
  
  // 应用关键词搜索
  if (options?.keyword) {
    const lowerKeyword = options.keyword.toLowerCase();
    collection = collection.filter(note =>
      note.title.toLowerCase().includes(lowerKeyword) ||
      note.content.toLowerCase().includes(lowerKeyword)
    );
  }
  
  // 应用时间筛选
  if (options?.timeRange) {
    const cutoffTime = this.calculateCutoffTime(options.timeRange);
    collection = collection.filter(note =>
      (note.archived_at || 0) >= cutoffTime
    );
  }
  
  // 应用分类筛选
  if (options?.categoryIds && options.categoryIds.length > 0) {
    collection = collection.filter(note =>
      options.categoryIds!.includes(note.category_id)
    );
  }
  
  // 排序
  collection = collection.sortBy('archived_at').reverse();
  
  // 分页
  if (options?.offset) {
    collection = collection.offset(options.offset);
  }
  if (options?.limit) {
    collection = collection.limit(options.limit);
  }
  
  return collection.toArray();
}

/**
 * 计算时间筛选截止点
 */
private calculateCutoffTime(timeRange: string): number {
  const now = Date.now();
  const daysMap: Record<string, number> = {
    '1d': 1,
    '7d': 7,
    '30d': 30,
    'all': 0
  };
  const days = daysMap[timeRange] || 0;
  return days > 0 ? now - days * 24 * 60 * 60 * 1000 : 0;
}

/**
 * 归档查询选项
 */
interface ArchiveQueryOptions {
  keyword?: string;
  timeRange?: string;
  categoryIds?: string[];
  offset?: number;
  limit?: number;
}
```
