## 文档信息
- **项目名称：** KINETIC_NOTES（MiniAI记事本）
- **模块名称：** 新闻模块（News Module）
- **文档版本：** v1.0

- **前端架构师：** Claude AI

## 1. 模块概述

### 1.1 功能定位

新闻模块作为 MiniAI 记事本的信息入口，提供精选新闻内容浏览、阅读和收藏功能。用户可快速浏览高质量资讯，必要时一键收藏到记事本进行深度记录。

### 1.2 设计风格遵循

严格遵循 Neo-Brutalist 设计系统规范：

| 设计参数 | 值 | 应用场景 |
|---------|-----|---------|
| 主背景色 | #131313 | 页面背景、卡片背景 |
| 刷新按钮 | #FFD700 | 高亮强调 |
| 卡片边框 | 4px #FFFFFF | 所有新闻卡片 |
| 卡片圆角 | 0px | 无圆角设计 |
| 卡片阴影 | 8px偏移 | Neo阴影效果 |
| Hover效果 | -translate-y-1 -translate-x-1 | 卡片交互 |

### 1.3 核心功能点

| 功能点 | 描述 | 优先级 |
|-------|------|--------|
| NewsList | Bento Grid布局、下拉刷新、上拉加载 | P0 |
| NewsCard | 标题、摘要、来源、时间展示 | P0 |
| NewsDetail | 全文展示、收藏按钮、返回导航 | P0 |

## 3. 数据模型设计

### 3.1 新闻实体定义

```typescript
// types/news.ts

/**
 * 新闻文章实体
 */
interface NewsArticle {
  // ========== 核心内容 ==========
  id: string                 // 新闻唯一ID（来自API）
  title: string              // 新闻标题 (≤100字符)
  summary: string            // 新闻摘要 (≤200字符)
  content: string            // 新闻正文全文
  cover_image?: string       // 封面图片URL
  
  // ========== 来源信息 ==========
  source: string             // 新闻来源名称
  source_url: string         // 来源原始URL
  author?: string            // 作者名称
  
  // ========== 时间信息 ==========
  published_at: number       // 发布时间戳
  crawled_at: number         // 抓取时间戳
  
  // ========== 分类标签 ==========
  category: NewsCategory     // 新闻分类
  tags: string[]             // 标签列表
  
  // ========== 统计信息 ==========
  view_count?: number        // 浏览次数（可选）
  
  // ========== 本地状态 ==========
  is_read: boolean           // 是否已读
  is_favorite: boolean       // 是否收藏
  favorite_at?: number       // 收藏时间
}

/**
 * 新闻分类枚举
 */
type NewsCategory = 
  | 'tech'       // 科技
  | 'finance'    // 财经
  | 'life'       // 生活
  | 'world'      // 国际
  | 'entertainment' // 娱乐

/**
 * 新闻列表响应
 */
interface NewsListResponse {
  articles: NewsArticle[]
  total: number
  page: number
  page_size: number
  has_more: boolean
}

/**
 * 新闻缓存实体（IndexedDB存储）
 */
interface NewsCacheEntity extends NewsArticle {
  // ========== 缓存元数据 ==========
  cached_at: number          // 缓存时间
  cache_expire_at: number    // 缓存过期时间
  cache_status: CacheStatus  // 缓存状态
}

type CacheStatus = 'fresh' | 'stale' | 'expired'

/**
 * 新闻收藏实体（与笔记系统关联）
 */
interface NewsFavorite {
  id: string                 // 收藏记录ID
  news_id: string            // 关联的新闻ID
  note_id?: string           // 关联的笔记ID（可选）
  created_at: number         // 收藏时间
}
```

### 3.2 IndexedDB Schema设计

```typescript
// repositories/newsRepository.ts

import Dexie, { Table } from 'dexie'

/**
 * 新闻模块数据库扩展
 */
class NewsDB extends Dexie {
  newsCache!: Table<NewsCacheEntity, string>
  newsFavorites!: Table<NewsFavorite, string>

  constructor() {
    super('MiniAINewsDB')
    
    this.version(1).stores({
      // 新闻缓存表
      newsCache: 'id, category, published_at, cached_at, cache_expire_at, [category+published_at]',
      
      // 新闻收藏表
      newsFavorites: 'id, news_id, created_at',
    })
  }
}

export const newsDb = new NewsDB()
```

### 3.3 Pinia Store设计

```typescript
// stores/newsStore.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { NewsArticle, NewsCategory } from '@/types/news'

export const useNewsStore = defineStore('news', () => {
  // ========== 状态定义 ==========
  const articles = ref<NewsArticle[]>([])
  const currentArticle = ref<NewsArticle | null>(null)
  const loading = ref(false)
  const refreshing = ref(false)
  const error = ref<string | null>(null)
  const page = ref(1)
  const hasMore = ref(true)
  const selectedCategory = ref<NewsCategory | 'all'>('all')
  
  // ========== 计算属性 ==========
  const filteredArticles = computed(() => {
    if (selectedCategory.value === 'all') {
      return articles.value
    }
    return articles.value.filter(a => a.category === selectedCategory.value)
  })
  
  const favoriteArticles = computed(() => {
    return articles.value.filter(a => a.is_favorite)
  })
  
  const unreadCount = computed(() => {
    return articles.value.filter(a => !a.is_read).length
  })
  
  // ========== 异步操作 ==========
  async function fetchNews(pageNum: number = 1) {
    loading.value = true
    error.value = null
    try {
      const response = await newsApi.getNews({
        page: pageNum,
        category: selectedCategory.value,
        pageSize: 10
      })
      
      if (pageNum === 1) {
        articles.value = response.articles
      } else {
        articles.value.push(...response.articles)
      }
      
      page.value = pageNum
      hasMore.value = response.has_more
      
      // 缓存到IndexedDB
      await newsCacheService.cacheArticles(response.articles)
    } catch (e) {
      error.value = e instanceof Error ? e.message : '加载失败'
      // 离线时从缓存读取
      await loadFromCache()
    } finally {
      loading.value = false
    }
  }
  
  async function refreshNews() {
    refreshing.value = true
    page.value = 1
    await fetchNews(1)
    refreshing.value = false
  }
  
  async function loadMore() {
    if (!hasMore.value || loading.value) return
    await fetchNews(page.value + 1)
  }
  
  async function loadFromCache() {
    const cached = await newsCacheService.getCachedArticles(selectedCategory.value)
    articles.value = cached
  }
  
  async function toggleFavorite(articleId: string) {
    const article = articles.value.find(a => a.id === articleId)
    if (!article) return
    
    article.is_favorite = !article.is_favorite
    article.favorite_at = article.is_favorite ? Date.now() : undefined
    
    await newsRepository.updateFavorite(articleId, article.is_favorite)
  }
  
  function setCategory(category: NewsCategory | 'all') {
    selectedCategory.value = category
    page.value = 1
    articles.value = []
    fetchNews(1)
  }
  
  function setCurrentArticle(article: NewsArticle) {
    currentArticle.value = article
    article.is_read = true
  }
  
  return {
    // 状态
    articles,
    currentArticle,
    loading,
    refreshing,
    error,
    page,
    hasMore,
    selectedCategory,
    
    // 计算属性
    filteredArticles,
    favoriteArticles,
    unreadCount,
    
    // 操作
    fetchNews,
    refreshNews,
    loadMore,
    loadFromCache,
    toggleFavorite,
    setCategory,
    setCurrentArticle,
  }
})
```

### 4.1 新闻列表组件 (NewsList)

**组件职责：**
- 展示新闻卡片Bento Grid布局
- 实现下拉刷新交互
- 实现上拉加载更多
- 处理加载状态和错误状态

**组件接口：**

```typescript
// components/NewsList.vue

interface NewsListProps {
  category?: NewsCategory | 'all'  // 筛选分类
  pageSize?: number                 // 每页数量，默认10
}

interface NewsListEmits {
  (e: 'article-click', article: NewsArticle): void  // 点击新闻卡片
  (e: 'refresh'): void                               // 刷新事件
  (e: 'load-more'): void                             // 加载更多事件
```

**组件实现：**

```vue
<!-- components/NewsList.vue -->
<template>
  <div class="news-list-container">
    <!-- 顶部刷新按钮 -->
    <NewsRefreshBtn 
      :loading="refreshing"
      @click="handleRefresh"
    />
    
    <!-- Bento Grid 新闻网格 -->
    <div 
      ref="gridRef"
      class="news-grid"
      :class="gridClass"
    >
      <NewsCard
        v-for="article in filteredArticles"
        :key="article.id"
        :article="article"
        :size="getCardSize(article)"
        @click="handleArticleClick(article)"
      />
      
      <!-- 大卡片占位（Bento Grid特性） -->
      <div 
        v-if="showFeaturedCard"
        class="featured-slot"
      >
        <NewsCard
          :article="featuredArticle"
          size="large"
          featured
        />
      </div>
    </div>
    
    <!-- 加载状态 -->
    <NewsLoadingState 
      v-if="loading"
      :type="loadingType"
    />
    
    <!-- 加载更多触发器 -->
    <div 
      ref="loadMoreTrigger"
      class="load-more-trigger"
    >
      <span v-if="!hasMore" class="no-more-text">
        NO MORE NEWS
      </span>
    </div>
    
    <!-- 错误状态 -->
    <div v-if="error" class="error-state">
      <NeoCard variant="error">
        <p class="error-message">{{ error }}</p>
        <NeoButton @click="handleRetry">RETRY</NeoButton>
      </NeoCard>
    </div>
    
    <!-- 下拉刷新指示器 -->
    <div 
      v-show="isPulling"
      class="pull-indicator"
      :style="{ transform: `translateY(${pullDistance}px)` }"
    >
      <div class="pull-icon">
        <IconRefresh :class="{ rotating: refreshing }" />
      </div>
      <span class="pull-text">{{ pullText }}</span>
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, computed, onMounted, onUnmounted } from 'vue'
import { useNewsStore } from '@/stores/newsStore'
import { useInfiniteScroll } from '@/composables/useNewsInfinite'
import { usePullRefresh } from '@/composables/useNewsRefresh'
import NewsCard from './NewsCard.vue'
import NewsRefreshBtn from './NewsRefreshBtn.vue'
import NewsLoadingState from './NewsLoadingState.vue'
import NeoCard from '@/components/base/NeoCard.vue'
import NeoButton from '@/components/base/NeoButton.vue'
import IconRefresh from '@/components/icons/IconRefresh.vue'
import type { NewsArticle, NewsCategory } from '@/types/news'

// Props
const props = withDefaults(defineProps<NewsListProps>(), {
  category: 'all',
  pageSize: 10,
})

// Emits
const emit = defineEmits<NewsListEmits>()

// Store
const newsStore = useNewsStore()

// 响应式状态
const gridRef = ref<HTMLElement | null>(null)
const loadMoreTrigger = ref<HTMLElement | null>(null)
const isPulling = ref(false)
const pullDistance = ref(0)

// 计算属性
const filteredArticles = computed(() => newsStore.filteredArticles)
const loading = computed(() => newsStore.loading)
const refreshing = computed(() => newsStore.refreshing)
const hasMore = computed(() => newsStore.hasMore)
const error = computed(() => newsStore.error)

const loadingType = computed(() => {
  return newsStore.page === 1 ? 'initial' : 'more'
})

const pullText = computed(() => {
  if (pullDistance.value > 80) return 'RELEASE TO REFRESH'
  return 'PULL TO REFRESH'
})

// Bento Grid布局响应式
const gridClass = computed(() => {
  return {
    'grid-cols-1': true,           // Mobile单列
    'md:grid-cols-2': true,        // Tablet双列
    'lg:grid-cols-3': true,        // Desktop三列
    'gap-4': true,
  }
})

// 大卡片逻辑（第一条新闻展示为大卡片）
const showFeaturedCard = computed(() => {
  return newsStore.page === 1 && filteredArticles.value.length > 0
})
const featuredArticle = computed(() => filteredArticles.value[0])

// 方法
function getCardSize(article: NewsArticle): 'normal' | 'large' {
  if (showFeaturedCard.value && article.id === featuredArticle.value?.id) {
    return 'large'
  }
  return 'normal'
}

function handleArticleClick(article: NewsArticle) {
  newsStore.setCurrentArticle(article)
  emit('article-click', article)
}

async function handleRefresh() {
  await newsStore.refreshNews()
  emit('refresh')
}

async function handleLoadMore() {
  await newsStore.loadMore()
  emit('load-more')
}

function handleRetry() {
  newsStore.fetchNews(1)
}

// 无限滚动逻辑
const { observe } = useInfiniteScroll(loadMoreTrigger, {
  onLoadMore: handleLoadMore,
  disabled: computed(() => !hasMore.value || loading.value),
})

// 下拉刷新逻辑
const { initPullRefresh, destroyPullRefresh } = usePullRefresh(gridRef, {
  onPullStart: () => { isPulling.value = true },
  onPull: (distance: number) => { pullDistance.value = distance },
  onPullEnd: () => { 
    isPulling.value = false
    pullDistance.value = 0
  },
  onRefresh: handleRefresh,
  threshold: 80,
})

// 生命周期
onMounted(() => {
  if (newsStore.articles.length === 0) {
    newsStore.fetchNews(1)
  }
  observe()
  initPullRefresh()
})

onUnmounted(() => {
  destroyPullRefresh()
})
</script>
```

### 4.2 新闻卡片组件 (NewsCard)

**组件职责：**
- 展示新闻标题、摘要、来源、时间
- 实现Neo-Brutalist视觉风格
- 处理卡片Hover交互效果
- 支持大卡片和小卡片两种尺寸

**组件接口：**

```typescript
// components/NewsCard.vue

interface NewsCardProps {
  article: NewsArticle        // 新闻数据
  size?: 'normal' | 'large'   // 卡片尺寸
  featured?: boolean          // 是否为特色大卡片
}

interface NewsCardEmits {
  (e: 'click', article: NewsArticle): void  // 点击卡片
}
```

**组件实现：**

```vue
<!-- components/NewsCard.vue -->
<template>
  <article
    class="news-card"
    :class="cardClass"
    @click="handleClick"
    @mouseenter="handleHoverEnter"
    @mouseleave="handleHoverLeave"
  >
    <!-- 卡片封面图（大卡片显示） -->
    <div 
      v-if="showCover && article.cover_image"
      class="card-cover"
    >
      <img 
        :src="article.cover_image" 
        :alt="article.title"
        class="cover-image"
        loading="lazy"
      />
      <!-- 来源标签 -->
      <NeoTag 
        :color="tagColor"
        :rotation="-2"
        class="source-tag"
      >
        {{ article.source }}
      </NeoTag>
    </div>
    
    <!-- 卡片内容 -->
    <div class="card-content">
      <!-- 标题 -->
      <h3 class="card-title">
        {{ article.title }}
      </h3>
      
      <!-- 摘要 -->
      <p 
        v-if="showSummary"
        class="card-summary"
      >
        {{ truncatedSummary }}
      </p>
      
      <!-- 元信息 -->
      <div class="card-meta">
        <!-- 来源（小卡片显示） -->
        <span 
          v-if="!showCover"
          class="meta-source"
        >
          {{ article.source }}
        </span>
        
        <!-- 发布时间 -->
        <span class="meta-time">
          {{ formattedTime }}
        </span>
        
        <!-- 已读/收藏状态 -->
        <div class="meta-status">
          <IconFavorite 
            v-if="article.is_favorite"
            class="status-icon favorite"
          />
          <IconUnread 
            v-if="!article.is_read"
            class="status-icon unread"
          />
        </div>
      </div>
    </div>
    
    <!-- Neo-Brutalist阴影层 -->
    <div class="neo-shadow-layer" :class="{ 'shadow-hover': isHovered }"></div>
  </article>
</template>

<script setup lang="ts">
import { ref, computed } from 'vue'
import NeoTag from '@/components/base/NeoTag.vue'
import IconFavorite from '@/components/icons/IconFavorite.vue'
import IconUnread from '@/components/icons/IconUnread.vue'
import type { NewsArticle } from '@/types/news'
import { formatRelativeTime } from '@/utils/newsFormatter'

// Props
const props = withDefaults(defineProps<NewsCardProps>(), {
  size: 'normal',
  featured: false,
})

// Emits
const emit = defineEmits<NewsCardEmits>()

// 响应式状态
const isHovered = ref(false)

// 计算属性
const cardClass = computed(() => ({
  'news-card--normal': props.size === 'normal',
  'news-card--large': props.size === 'large',
  'news-card--featured': props.featured,
  'is-read': props.article.is_read,
}))

const showCover = computed(() => props.size === 'large' || props.featured)
const showSummary = computed(() => props.size === 'large' || props.featured)

const truncatedSummary = computed(() => {
  const maxLen = props.size === 'large' ? 150 : 100
  if (props.article.summary.length <= maxLen) {
    return props.article.summary
  }
  return props.article.summary.slice(0, maxLen) + '...'
})

const formattedTime = computed(() => {
  return formatRelativeTime(props.article.published_at)
})

const tagColor = computed(() => {
  // 根据分类返回颜色
  const colorMap = {
    tech: 'yellow',
    finance: 'cyan',
    life: 'white',
    world: 'gray',
    entertainment: 'yellow',
  }
  return colorMap[props.article.category] || 'yellow'
})

// 方法
function handleClick() {
  emit('click', props.article)
}

function handleHoverEnter() {
  isHovered.value = true
}

function handleHoverLeave() {
  isHovered.value = false
}
</script>
```

```typescript
// composables/useNewsInfinite.ts

import { ref, onMounted, onUnmounted, type Ref } from 'vue'

interface InfiniteScrollOptions {
  onLoadMore: () => Promise<void> | void
  disabled: Ref<boolean>
  threshold?: number  // 触发距离，默认100px
}

export function useInfiniteScroll(
  triggerRef: Ref<HTMLElement | null>,
  options: InfiniteScrollOptions
) {
  const { onLoadMore, disabled, threshold = 100 } = options
  const observer = ref<IntersectionObserver | null>(null)
  const isLoading = ref(false)

  function observe() {
    if (!triggerRef.value) return

    observer.value = new IntersectionObserver(
      async (entries) => {
        const entry = entries[0]
        
        if (entry.isIntersecting && !disabled.value && !isLoading.value) {
          isLoading.value = true
          try {
            await onLoadMore()
          } finally {
            isLoading.value = false
          }
        }
      },
      {
        rootMargin: `${threshold}px`,
        threshold: 0,
      }
    )

    observer.value.observe(triggerRef.value)
  }

  function disconnect() {
    if (observer.value) {
      observer.value.disconnect()
      observer.value = null
    }
  }

  onUnmounted(() => {
    disconnect()
  })

  return {
    observe,
    disconnect,
    isLoading,
  }
}
```

### 5.2 下拉刷新逻辑 (useNewsRefresh)

```typescript
// composables/useNewsRefresh.ts

import { ref, onMounted, onUnmounted, type Ref } from 'vue'

interface PullRefreshOptions {
  onPullStart: () => void
  onPull: (distance: number) => void
  onPullEnd: () => void
  onRefresh: () => Promise<void> | void
  threshold?: number  // 刷新触发距离，默认80px
}

export function usePullRefresh(
  containerRef: Ref<HTMLElement | null>,
  options: PullRefreshOptions
) {
  const { onPullStart, onPull, onPullEnd, onRefresh, threshold = 80 } = options
  
  const isPulling = ref(false)
  const startY = ref(0)
  const currentY = ref(0)
  const isRefreshing = ref(false)

  function handleTouchStart(e: TouchEvent) {
    if (containerRef.value?.scrollTop === 0) {
      startY.value = e.touches[0].clientY
      isPulling.value = true
      onPullStart()
    }
  }

  function handleTouchMove(e: TouchEvent) {
    if (!isPulling.value) return
    
    currentY.value = e.touches[0].clientY
    const distance = Math.max(0, currentY.value - startY.value)
    
    onPull(distance)
  }

  function handleTouchEnd() {
    if (!isPulling.value) return
    
    const distance = currentY.value - startY.value
    
    if (distance > threshold && !isRefreshing.value) {
      isRefreshing.value = true
      onRefresh().finally(() => {
        isRefreshing.value = false
      })
    }
    
    isPulling.value = false
    startY.value = 0
    currentY.value = 0
    onPullEnd()
  }

  function initPullRefresh() {
    if (!containerRef.value) return
    
    const container = containerRef.value
    container.addEventListener('touchstart', handleTouchStart, { passive: true })
    container.addEventListener('touchmove', handleTouchMove, { passive: true })
    container.addEventListener('touchend', handleTouchEnd)
  }

  function destroyPullRefresh() {
    if (!containerRef.value) return
    
    const container = containerRef.value
    container.removeEventListener('touchstart', handleTouchStart)
    container.removeEventListener('touchmove', handleTouchMove)
    container.removeEventListener('touchend', handleTouchEnd)
  }

  return {
    initPullRefresh,
    destroyPullRefresh,
    isRefreshing,
  }
}
```

```typescript
import type { NewsArticle, NewsListResponse, NewsCategory } from '@/types/news'

interface NewsApiConfig {
  baseUrl: string
  apiKey?: string
  timeout: number
}

const defaultConfig: NewsApiConfig = {
  baseUrl: '/api/news',  // 可配置为实际新闻API

class NewsApiService {
  private config: NewsApiConfig

  constructor(config: NewsApiConfig = defaultConfig) {
    this.config = config
  }

  /**
   * 获取新闻列表
   */
  async getNews(params: {
    page: number
    category?: NewsCategory | 'all'
    pageSize: number
  }): Promise<NewsListResponse> {
    const queryParams = new URLSearchParams({
      page: params.page.toString(),
      pageSize: params.pageSize.toString(),
    })
    
    if (params.category && params.category !== 'all') {
      queryParams.set('category', params.category)
    }

    const response = await this.fetchWithTimeout(
      `${this.config.baseUrl}?${queryParams}`
    )

    if (!response.ok) {
      throw new Error(`API Error: ${response.status}`)
    }

    return response.json()
  }

  /**
   * 获取新闻详情
   */
  async getNewsDetail(id: string): Promise<NewsArticle> {
    const response = await this.fetchWithTimeout(
      `${this.config.baseUrl}/${id}`
    )

    if (!response.ok) {
      throw new Error(`API Error: ${response.status}`)
    }

    return response.json()
  }

  /**
   * 带超时的fetch封装
   */
  private async fetchWithTimeout(url: string): Promise<Response> {
    const controller = new AbortController()
    const timeoutId = setTimeout(
      () => controller.abort(),
      this.config.timeout
    )

    try {
      const response = await fetch(url, {
        signal: controller.signal,
        headers: this.config.apiKey
          ? { 'Authorization': `Bearer ${this.config.apiKey}` }
          : {},
      })
      return response
    } finally {
      clearTimeout(timeoutId)
    }
  }
}
```

```typescript
// services/newsCache.ts

import { newsDb } from '@/repositories/newsRepository'
import type { NewsArticle, NewsCategory, NewsCacheEntity } from '@/types/news'

const CACHE_DURATION = 30 * 60 * 1000  // 30分钟缓存有效期

class NewsCacheService {
  /**
   * 缓存新闻文章
   */
  async cacheArticles(articles: NewsArticle[]): Promise<void> {
    const now = Date.now()
    const cacheEntities: NewsCacheEntity[] = articles.map(article => ({
      ...article,
      cached_at: now,
      cache_expire_at: now + CACHE_DURATION,
      cache_status: 'fresh',
    }))

    await newsDb.newsCache.bulkPut(cacheEntities)
  }

  /**
   * 获取缓存的新闻文章
   */
  async getCachedArticles(category?: NewsCategory | 'all'): Promise<NewsArticle[]> {
    const now = Date.now()
    
    let query = newsDb.newsCache
      .where('cache_expire_at')
      .above(now)
    
    if (category && category !== 'all') {
      query = query.and(entity => entity.category === category)
    }

    const cached = await query
      .reverse()
      .sortBy('published_at')

    return cached.map(entity => this.entityToArticle(entity))
  }

  /**
   * 获取单条缓存新闻
   */
  async getCachedArticle(id: string): Promise<NewsArticle | null> {
    const entity = await newsDb.newsCache.get(id)
    return entity ? this.entityToArticle(entity) : null
  }

  /**
   * 清理过期缓存
   */
  async clearExpiredCache(): Promise<void> {
    const now = Date.now()
    await newsDb.newsCache
      .where('cache_expire_at')
      .below(now)
      .delete()
  }

  /**
   * 更新缓存状态
   */
  async updateCacheStatus(id: string, status: 'fresh' | 'stale' | 'expired'): Promise<void> {
    await newsDb.newsCache.update(id, { cache_status: status })
  }

  /**
   * 实体转文章对象
   */
  private entityToArticle(entity: NewsCacheEntity): NewsArticle {
    return {
      id: entity.id,
      title: entity.title,

      content: entity.content,
      cover_image: entity.cover_image,
      source: entity.source,
      source_url: entity.source_url,
      author: entity.author,
      published_at: entity.published_at,
      crawled_at: entity.crawled_at,
      category: entity.category,
      tags: entity.tags,
      view_count: entity.view_count,
      is_read: entity.is_read,
      is_favorite: entity.is_favorite,
      favorite_at: entity.favorite_at,
    }
  }
}
```

```typescript
// utils/newsFormatter.ts

/**
 * 格式化相对时间（如：2小时前、3天前）
 */
export function formatRelativeTime(timestamp: number): string {
  const now = Date.now()
  const diff = now - timestamp
  
  const minutes = Math.floor(diff / (60 * 1000))
  const hours = Math.floor(diff / (60 * 60 * 1000))
  const days = Math.floor(diff / (24 * 60 * 60 * 1000))
  
  if (minutes < 1) return 'JUST NOW'
  if (minutes < 60) return `${minutes} MIN AGO`
  if (hours < 24) return `${hours} HOUR AGO`
  if (days < 7) return `${days} DAY AGO`
  
  // 超过7天显示具体日期
  return formatDate(timestamp)
}

/**
 * 格式化绝对日期（如：2026-04-20）
 */
export function formatDate(timestamp: number): string {
  const date = new Date(timestamp)
  const year = date.getFullYear()
  const month = String(date.getMonth() + 1).padStart(2, '0')
  const day = String(date.getDate()).padStart(2, '0')
  
  return `${year}-${month}-${day}`
}

/**
 * HTML内容安全处理
 */
export function sanitizeHtml(html: string): string {
  // 移除潜在危险的标签和属性
  // 实际生产环境建议使用 DOMPurify 库
  
  const dangerousTags = ['script', 'iframe', 'object', 'embed', 'form']
  const dangerousAttrs = ['onclick', 'onload', 'onerror', 'onmouseover']
  
  let sanitized = html
  
  // 移除危险标签
  dangerousTags.forEach(tag => {
    const regex = new RegExp(`<${tag}[^>]*>.*?</${tag}>`, 'gi')
    sanitized = sanitized.replace(regex, '')
  })
  
  // 移除危险属性
  dangerousAttrs.forEach(attr => {
    const regex = new RegExp(`${attr}="[^"]*"`, 'gi')
    sanitized = sanitized.replace(regex, '')
  })
  
  return sanitized
}
```
